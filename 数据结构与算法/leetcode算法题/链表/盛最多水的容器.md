# 双指针

```
func maxArea(height []int) int {
    l, r := 0, len(height) - 1
    ans := 0
    for l < r {
        area := (r - l) * min(height[l], height[r])
        ans = max(ans, area)
        if height[l] < height[r] {
            l++
        } else {
            r--
        }
    }

    return ans
}

func max(a, b int) int {
    if a > b {
        return a
    }

    return b
}

func min(a, b int) int {
    if a < b {
        return a
    }

    return b
}
```
